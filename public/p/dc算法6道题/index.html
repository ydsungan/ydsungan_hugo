function partition(A, l, r)
1: Choose an index x from [l,r] at random and use A[x] as pivot
2: Swap A[x] with A[r]
3: pivot = A[r]; i = l;
4: for j = l to r - 1 do
5:     if A[j] < pivot then
6:         Swap A[i] with A[j];
7:         i ++;
8:     end if

2: initialize two list node: head, tail;
